## 一、01背包问题

区分：每件物品最多只能选一次

示例代码：

```c++
for(int i = 1; i <= n ; i ++)

    for(int j = m ; j >= v[i]; j --) // 注意：从后往前遍历，括号内的dp[j]和dp[j-v[i]]为i-1的情况下的最大值

        dp[j] = max(dp[j], dp[ j - v[i] ] + w[ i ] );

ans = dp[ m ];
```

## 二、完全背包问题

区分：每件物品可选无限次

示例代码：

```c++
for(int i = 1 ; i <= n ; i ++)

    for(int j = v[i] ; j <= m ; j ++) // 注意：从前往后遍历，使得可以多次使用当前物品

        dp[j] = max(dp[j], dp[ j-v[i] ] + w[ i ]);

ans = dp[ m ];
```



## 三、多重背包问题（转化成01背包）

区分：每件物品的数量有限

示例代码：

```c++
        for(int i = 1; i <= n; i ++)

            for(int j = m; j >= 0; j --)

                for(int k = 1; k <= a[i] && k * v[i] <= j; k ++) // 注意：等于01背包问题只能选择0和1，转变为只能选择0~a[i]个

                    dp[j] = max(dp[j], dp[j-k * v[i]] + k * w[i]);

        ans = dp[m];
```

二进制优化：

```c++
struct Goods{
    int v,w;
};

vector<Goods> goods;

// 对每一种物品进行二进制个数拆分，将其转化成01背包问题
for(int i = 1; i <= n; i ++)
    for(int k = 1; k <= a[i]; k * = 2)
        a[i] -= k;
        goods.push_back({k * v[i], k * w[i]});
    if(a[i] > 0) goods.push_back({a[i] * v[i], a[i] * w[i]});

for(auto &g : goods)
    for(int j = m; j >= g.v; j --)
        dp[j] = max(dp[j], dp[j-g.v] + g.w);

```

## 四、混合背包问题（分成01背包和完全背包）

区分：每种物品分为最多可选一个，无限个和有限个三种情况。

示例代码：

```c++
struct Thing{
    int kind,v,w;
};

vector<Thing> things;

// 输入
for(int i = 0; i < n; i ++)
{
    int s, v, w;
    cin >> s >> v >> w;
    if(s < 0) things.push_back({-1, v, w});
    else if(s == 0) things.push_back({0, v, w});
    else 
    {
        // 二进制拆分成01背包
        for(int k = 1; k <= s; k *= 2)
        {
            s -= k;
            things.push_back({-1, k * v, k * w});
        }
        if(s > 0) things.push_back({-1, s * v, s * w});
    }
}
// 遍历
for(auto &p: things)
{
    if(p.kind == -1) // 01背包
    {
        for(int j = m; j >= p.v; j --)
            dp[j]=max(dp[j], dp[j - p.v] + p.w);
    }
    else // 完全背包
    {
        for(int j = p.v; j <= m; j ++)
            dp[j]=max(dp[j], dp[j - p.v] + p.w);
    }
}

ans = dp[m];
```

## 五、二维费用的背包问题

区分：固定背包的体积和背包的重量。

示例代码：

```c++
for(int i = 0; i < n; i ++)
{
    int a, b, c;
    cin >> a >> b >> c; // a：体积，b：重量，c：价值
    for(int j = v; j >= a; j --)
        for(int k = m; k >= b; k --)
            dp[j][k] = max(dp[j][k], dp[j - a][k - b] + c);
}

ans = dp[v][m];
```

## 六、分组背包问题

区分：所有物品分为若干组，每组中只能选择一种物品。

示例代码：

```c++
int v[N], w[N];
for(int i = 0; i < n; i ++) // n组物品
{
    int s; // s种物品
    cin >> s;
    for(int k = 1; k <= s; k ++) cin >> v[k] >> w[k];
    for(int j = m; j >= 0; j --)
        for(int k = 0; k <= s; k ++) // 对于每组物品，有（n+1）中选择方式，类似多重背包问题
            if(j >= v[k])
                dp[j] = max(dp[j], dp[j - v[i]] + w[i]);
}

ans = dp[m];
```
