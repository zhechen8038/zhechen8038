## 代码示例：

```c++
vector<int> prefix_function(string s){
    int n = s.size();
    vector<int> pi(n); // 匹配串的最大长度

    pi[0] = 0; // 第一个的匹配串长度为0
    for(int i = 1; i < n; i ++)
    {
        int j = pi[i - 1]; // 当前匹配串的最大长度为上一个匹配串长度加1
        while(j > 0 && s[j] != s[i]) j = pi[j - 1]; // 往前递归匹配串的最大长度
        if(s[j] == s[i]) j ++; // 匹配成功，则在最大长度的基础上加1
        pi[i] = j;
    }
    return pi;
}
```

# 二、KMP算法

## 问题：在字符串中查找子串出现的位置

## 关键：利用前缀函数求解

![1e6baaa2-4fce-49f9-9760-7d314519b174](file:///C:/Users/HP/Pictures/Typedown/1e6baaa2-4fce-49f9-9760-7d314519b174.png)

## 代码示例：

```c++
vector<int> find_occurrences(string text, string pattern){
    string cur = pattern + "#" + text; // 拼接
    int n1 = pattern.size(), n2 = text.size();
    vector<int> v; // 存储出现的位置
    vector<int> lps = prefix_function(cur); // 求前缀函数值
    for(int i = n1 + 1; i < n1 + n2 + 1; i ++)
    {
        if(lps[i] == n1) v.push_back(i - (n1 - 1) - (n1 + 1)); // 取相对位置
    }
    return v;
}
```

# 三、字典树（Trie）

## ![a2910b70-7434-49b6-8f6c-309705a251a9](file:///C:/Users/HP/Pictures/Typedown/a2910b70-7434-49b6-8f6c-309705a251a9.png)

## 原理：从根节点到任意一个节点代表一个字符串。

## 代码示例：

```c++
struct trie{
    int nex[100000][26], cnt; // nex[i][0]代表第i个节点指向下一个节点的有向线段上的字母为i+'a'，cnt代表当前树上的节点个数
    bool exist[100000] // 该结点结尾的字符串是否存在

    void insert(char *s, int l){
        int p = 0;
        for(int i = 0; i < l; i ++){
            int c = s[i] - 'a';
            if(!nex[p][c]) nex[p][c] = ++cnt; // 如果没有，就添加结点
            p = nex[p][c];
        }
        exist[p] = true; // 该字符串存在
    }

    int find(char *s, int l){
        int p = 0;
        for(int i = 0; i < l; i ++){
            int c = s[i] - 'a';
            if(!nex[p][c]) return 0;
            p = nex[p][c];
        }
        return exist[p];
    }
};
```

# 四、Manacher算法

## 解决问题：寻找最长回文串，或者计算以每一个点为中心的奇数长度回文串和偶数（以靠近右半边为中心）长度回文串。

## 思路：

![373f6a12-bcf4-47a9-8e7d-1bb11a8b0dcc](file:///C:/Users/HP/Pictures/Typedown/373f6a12-bcf4-47a9-8e7d-1bb11a8b0dcc.png)

![c2deb194-b879-4e8e-ae5c-02c22f63da8c](file:///C:/Users/HP/Pictures/Typedown/c2deb194-b879-4e8e-ae5c-02c22f63da8c.png)

## 简要概述：在查找当前节点的最长回文串时，利用回文串的对称性，如果该节点落在一个回文串的[L,R]中，那么该节点在这个回文串中的回文部分和其左对称点的回文部分相同，计算时就从对称部分长度开始，从而缩小检索范围。

## 代码示例：

（1）计算奇数长度串

```c++
vector<int> d1(n);
for(int i = 0, l = 0, r = -1; i < n; i ++){
    int k = (i > r) ? 1 : min(d1[l + r - i], r - i + 1); // 不能超过r的边界
    while(0 <= i - k && i + k < n; && s[i - k] == s[i + k]){
        k ++;
    } 
    d1[i] = k --; // 不计算本身
    if(i + k > r){ // 更新l,r
        l = i - k;
        r = i + k;    
    }
}
```

（2）计算偶数长度串

```c++
vector<int> d2(n);
for(int i = 0, l = 0, r = -1; i < n; i ++){
    int k = (i > r) ? 0 : min(d1[l + r - i + 1], r - i + 1); // 不能超过r的边界
    while(0 <= i - k - 1 && i + k < n; && s[i - k - 1] == s[i + k]){
        k ++;
    } 
    d2[i] = k --; // 不计算本身
    if(i + k > r){ // 更新l,r
        l = i - k - 1;
        r = i + k;    
    }
}
```
